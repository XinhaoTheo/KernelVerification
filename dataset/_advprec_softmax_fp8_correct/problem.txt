import torch
import torch.nn as nn

class Model(nn.Module):
    """Softmax with an FP8 output, CORRECT kernel. Pairs with the buggy
    _advprec_softmax_fp8_tail to show `abstain` is driven by the FORMAT, not the bug:
    even a correct low-bit kernel can't be judged numerically -> abstain, not match."""
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.softmax(x, dim=-1).to(torch.float8_e4m3fn)

num_features = 256

def get_inputs():
    return [torch.randn(8, num_features)]

def get_init_inputs():
    return []
