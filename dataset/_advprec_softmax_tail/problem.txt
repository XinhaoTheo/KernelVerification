import torch
import torch.nn as nn

class Model(nn.Module):
    """Softmax over the last dim — the precision_verification.md class-1 case.

    softmax compresses small (tail) values toward 0, so an error in the tail can
    be invisible in the output on a peaked distribution (false accept), while the
    same kernel breaks on a flat / heavy-tailed distribution where the tail
    carries real probability mass.
    """
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.softmax(x, dim=-1)

num_features = 256

def get_inputs():
    x = torch.randn(8, num_features)
    return [x]

def get_init_inputs():
    return []
