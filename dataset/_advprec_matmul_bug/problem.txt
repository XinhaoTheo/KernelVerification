import torch
import torch.nn as nn

class Model(nn.Module):
    """Square matmul x @ xᵀ — PRESERVE class. This entry ships a BUGGY kernel, but
    precision_recheck still returns `skipped`: preserve has no compress/select blind
    spot, so a preserve bug is the STANDARD recheck's job, not the precision axis's.
    Documents the boundary (the precision axis does not overreach)."""
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.matmul(x, x.transpose(-1, -2))

def get_inputs():
    return [torch.randn(16, 32)]

def get_init_inputs():
    return []
