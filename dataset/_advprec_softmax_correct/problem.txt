import torch
import torch.nn as nn

class Model(nn.Module):
    """Softmax over the last dim — same problem as _advprec_softmax_tail, but this
    entry ships a CORRECT kernel. It is a negative control: the precision battery
    must NOT flag it (no false alarm) even on the heavy-tail adversarial input."""
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.softmax(x, dim=-1)

num_features = 256

def get_inputs():
    return [torch.randn(8, num_features)]

def get_init_inputs():
    return []
