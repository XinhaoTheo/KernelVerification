import torch
import torch.nn as nn

class Model(nn.Module):
    """Square matmul with an FP8 output. The op is PRESERVE, but the low-bit output
    dominates: precision_recheck must `abstain` (low-bit numerical comparison is
    unreliable) — showing the abstain rule applies across op_classes, not just compress."""
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.matmul(x, x.transpose(-1, -2)).to(torch.float8_e4m3fn)

def get_inputs():
    return [torch.randn(16, 32)]

def get_init_inputs():
    return []
