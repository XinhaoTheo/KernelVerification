import torch
import torch.nn as nn

class Model(nn.Module):
    """Top-k key selection (indices) — same problem as _advprec_topk_boundary, but the
    buggy kernel uses a DIFFERENT mechanism (a strided subsample instead of a per-block
    quota). Second select catch, for mechanism diversity."""
    def __init__(self, k):
        super().__init__()
        self.k = k

    def forward(self, scores: torch.Tensor) -> torch.Tensor:
        return torch.topk(scores, self.k, dim=-1).indices

k = 8
num_keys = 256

def get_inputs():
    return [torch.randn(16, num_keys)]

def get_init_inputs():
    return [k]
