import torch
import torch.nn as nn

class Model(nn.Module):
    """Softmax with an FP8 output — precision_verification.md class 3 (softmax+FP8).

    The output is stored in float8_e4m3fn. FP8's smallest representable magnitude
    is ~0.002, so tail probabilities (~1e-3 and below) flush to 0. Any error that
    lives in the tail is therefore UNREPRESENTABLE in the output: a tail-wrong
    kernel produces a bit-identical FP8 tensor, error exactly 0. No tolerance and
    no input distribution can recover it -> the numerical judge must ABSTAIN
    (route to a downstream / task-level check), not report PASS.
    """
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.softmax(x, dim=-1).to(torch.float8_e4m3fn)

num_features = 256

def get_inputs():
    x = torch.randn(8, num_features)
    return [x]

def get_init_inputs():
    return []
