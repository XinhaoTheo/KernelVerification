import torch
import torch.nn as nn

class Model(nn.Module):
    """Square matmul x @ xᵀ — a PRESERVE-class op (magnitude-preserving). Negative
    control: precision_recheck has no compression/selection blind spot here, so it
    must return `skipped` (the standard recheck battery covers preserve correctness)."""
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.matmul(x, x.transpose(-1, -2))

def get_inputs():
    return [torch.randn(16, 32)]

def get_init_inputs():
    return []
