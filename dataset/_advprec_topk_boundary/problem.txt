import torch
import torch.nn as nn

class Model(nn.Module):
    """Top-k key selection — the `I = top-k(...)` step of sparse attention.

    Returns the INDICES of the k highest-scoring keys (a discrete selection, the
    kind of output that has no numerical distance). This is the class-2 case from
    precision_verification.md: |cand - ref| is undefined on an index set, so the
    numerical-error judge does not apply and J2 (value-gap / downstream) is needed.
    """
    def __init__(self, k):
        super().__init__()
        self.k = k

    def forward(self, scores: torch.Tensor) -> torch.Tensor:
        values, indices = torch.topk(scores, self.k, dim=-1)
        return indices

k = 8
num_keys = 256

def get_inputs():
    scores = torch.randn(16, num_keys)
    return [scores]

def get_init_inputs():
    return [k]
