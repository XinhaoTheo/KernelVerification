import torch
import torch.nn as nn

class Model(nn.Module):
    """Softmax — same as _advprec_softmax_tail, but the buggy kernel uses a DIFFERENT
    mechanism (a fixed COUNT cap instead of a value cutoff). Second compress catch,
    for mechanism diversity."""
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.softmax(x, dim=-1)
